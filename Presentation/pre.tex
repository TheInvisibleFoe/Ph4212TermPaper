    \section{Scovil Schulz-DuBois(SSD) Model}

    \begin{frame}{Scovil Schulz-DuBois(SSD) Model}
        \begin{itemize}
            \item The SSD model describes a three-level system that can operate as a quantum heat engine. \pause
            \item The three levels are denoted as $\ket{1}$, $\ket{2}$, and $\ket{3}$, with energies $E_1 < E_2 < E_3$. \pause
            \item The system is coupled to two thermal reservoirs: a hot reservoir at temperature $T_h$ and a cold reservoir at temperature $T_c$. \pause
            \item The engine operates by absorbing heat from the hot reservoir, performing work, and releasing heat to the cold reservoir.
        \end{itemize}
    \end{frame}

    \begin{frame}
        \frametitle{The Model}
        \begin{itemize}
            \item The SSD considers a three-level system with energy levels $E_1 < E_2 < E_3$. \pause
            \item The characteristic frequencies of the transitions are defined as:
            \begin{align*}
                \omega_{31} &= \omega_h = \frac{E_3 - E_1}{\hbar} \\
                \omega_{21} &= \omega_c = \frac{E_2 - E_1}{\hbar} \\ 
                \omega_{32} &= \nu = \frac{E_3 - E_2}{\hbar}
            \end{align*} \pause
            \item The system is coupled to two thermal reservoirs: a hot reservoir at temperature $T_h$ and a cold reservoir at temperature $T_c$. \pause
            \item The transition $\ket{3} \leftrightarrow \ket{1}$ is coupled to the hot reservoir, while the transition $\ket{2} \leftrightarrow \ket{1}$ is coupled to the cold reservoir. 
        \end{itemize}
    \end{frame} 

    \begin{frame}
        \frametitle{The Model}
        \begin{itemize}
            \item There is a filter that allows only frequencies near the vicinity of $\omega_{h}$ to be absorbed from the hot reservoir and only frequencies near the vicinity of $\omega_{c}$ to be absorbed by the cold reservoir. \pause
            \item In this model, for every photon of frequency $\omega_{h}$ absorbed from the hot reservoir, a photon of frequency $\omega_{c}$ is emitted into the cold reservoir, and a photon of frequency $\omega_{32}$ is emitted as work. \pause
            \item The engine operates in a cycle, where the system absorbs heat from the hot reservoir(allowing transitions with frequency $\omega_{h}$), performs work(emission with frequency $\omega_{32}$), and releases heat(allowing transitions with frequency $\omega_{c}$) to the cold reservoir.
        \end{itemize}
    \end{frame}

    \begin{frame}
        \frametitle{Population Distribution}
        \begin{itemize}
            \item At equilibrium, the populations of the energy levels are given by the Boltzmann distribution:
            \[N_i \propto \frac{e^{-E_i/k_B T}}{Z}\]
            where $k_B$ is the Boltzmann constant. \pause
            \item As a result, the population ratios are:
            \begin{align*}
                \frac{N_3}{N_1} = e^{-(E_3 - E_1)/k_B T_h} = e^{-\hbar \omega_h/k_B T_h} ;
                \frac{N_2}{N_1} = e^{-(E_2 - E_1)/k_B T_c} = e^{-\hbar \omega_c/k_B T_c}
            \end{align*} \pause
            \item For stimulated emission to occur, we require a population inversion, which means that the population of the higher energy level must be greater than that of the lower energy level. 
        \end{itemize}

    \end{frame}
    \begin{frame}
        \frametitle{Population Inversion and Efficiency}
        \begin{tcolorbox}[colback=red!5!white,colframe=red!75!black,title= Efficiency of the Engine]
            Similar to a classical heat engine, the efficiency of the SSD engine is given by:
            \begin{align*}
                \eta = \frac{\text{Work Output}}{\text{Heat Input}} = \frac{\hbar \nu}{\hbar \omega_{h}}
            \end{align*} 
        \end{tcolorbox}



        The condition for population inversion in the SSD model is given by:
        \begin{align*}
            N_3 > N_2 \implies  N_3/N_1 > N_2/N_1 .
        \end{align*}
    \end{frame}
